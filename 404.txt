    const domain = window.location.hostname;
    const lang = navigator.language || navigator.userLanguage;

    // daftar bahasa
    const texts = {
      id: {
        title: "Halaman Tidak Ditemukan",
        message: `Sepertinya halaman yang Anda cari di ${domain} tidak ditemukan.`,
        button: "Kembali ke Beranda"
      },
      en: {
        title: "Page Not Found",
        message: `The page you are looking for on ${domain} could not be found.`,
        button: "Back to Home"
      },
      fr: {
        title: "Page Introuvable",
        message: `La page que vous recherchez sur ${domain} est introuvable.`,
        button: "Retour à l'accueil"
      }
    };

    // ambil bahasa utama (misalnya "id-ID" -> "id")
    const userLang = lang.split("-")[0];

    // pilih bahasa, fallback ke en
    const t = texts[userLang] || texts["en"];

    document.getElementById("title").innerText = t.title;
    document.getElementById("message").innerText = t.message;
    document.getElementById("button").innerText = t.button;
